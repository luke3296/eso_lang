BNF - plain

step 1 arithmatic

<digit> ::= [0-9]
<character> ::== [a-zA-Z]
<int> ::= <digit>|<int><digit>

<arithmetic-expression> ::=   <int> put with <int>
			  |   <int> take from <int>
			  |   <int> by count of <int>
			  |   <int> by count per <int>

step 2 other stuff

<string> ::= <character> <string>
           | <character> <int>

<statement> ::=  <assignment-statment>
	     |  <if-statement>
	     |  <while-statement>
	     |  <for-statement>

<statements> ::= <statement> <statement>


<expression> ::=  <arithmetic-expression>
	       |  <relational-expression>


<relational-expression> ::=  <int> smaller than <int>
			  |  <int> bigger than <int>
			  |  <int> smaller or of a count with <int>
			  |  <int> bigger or of a count with <int>
			  |  <int> of a count with <int>
			  |  <int> of a count or bigger than <int>
			  |  <int> of a count or smaller than <int>
				|  <int> nae bigger than <int>
				|  <int> nae smaller than <int>
				|  <int> nae of a count with <int>


<varible> ::= <charactecter> <string>


<assignment-statement>  ::=   <varible> = <int>
	                |   <varible> = <string>



<if-statement> ::= Should the holde carry <variable> ; <relational-statement> ; <variable> make ready for <statements> then drink our rum!
		 | Mayhaps it Should carry <variable> ; <relational-expression> ; <variable> make ready for <statement> then drink our rum!
		 | Mayhaps <statements> then drink our rum!

<while-statement> ::= When on watch with <relational-expression> keep an eye open and <statements> then drink our rum!

<for-statement>   ::= On each day that <assignment-statement> ; <relational-expression> ; <statements> ye shall be sure to <statements> then drink our rum!


<function>  ::=  define <string> {<statements>}
	      |      print( <string> )
